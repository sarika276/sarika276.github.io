      </div>
      <div class="column">
        <input type="number" id="onCount" readonly placeholder="TOTAL LED_ON COUNT">
      </div>
    </div>

    <!-- Row 2 -->
    <div class="row">
      <div class="column">
        <button class="off-btn" onclick="incrementOff()">LED OFF</button>
      </div>
      <div class="column">
        <input type="number" id="offCount" readonly placeholder="TOTAL LED_OFF COUNT">
      </div>
    </div>

    <!-- Row 3 -->
    <div class="row">
      <div class="column" style="flex: 1;">
        <button class="reset-btn" onclick="resetCounts()">RESET</button>
      </div>
    </div>
  </div>

  <script>
    let onCounter = 0;
    let offCounter = 0;

    function incrementOn() {
      onCounter++;
      document.getElementById("onCount").value = onCounter;
    }

    function incrementOff() {
      offCounter++;
      document.getElementById("offCount").value = offCounter;
    }

    function resetCounts() {
      onCounter = 0;
      offCounter = 0;
      document.getElementById("onCount").value = "";
      document.getElementById("offCount").value = "";
    }
  </script>
